\begin{newmondai}{例題}{5}{今週の積分}{***}
    \int \frac{1}{\sin x} dx
\end{newmondai}

\approachitem{$f(\cos x)\cdot \sin x \rightarrow \cos x = t \text{と置換}$}

\solutionhead
\hfill
以下に計算手順を示す.
\hfill\
\begin{align*}
    \int \frac{1}{\sin x} dx
        &= \int \frac{\sin x}{\sin^2 x} dx \\
        &= \int \frac{\sin x}{1-\cos^2 x} dx
\end{align*}
\Itembox{
    t=\cos x \qquad 
    dt=-\sin x dx
}
\begin{align*}
    %\int \frac{1}{\sin x} dx
        &= \int \frac{1}{t^2-1} dt \\
        &= \int \frac{1}{(t-1)(t+1)} dt \\
        &= \frac{1}{2}\int \left(\frac{1}{t-1}-\frac{1}{t+1}\right) dt \\
        &= \frac{1}{2} \log \left|\frac{t-1}{t+1}\right|+\mathbb{C}  \\
        &= \boldsymbol{\frac{1}{2} \log \frac{1-\cos x}{1+\cos x}+\mathbb{C}}
\end{align*}
