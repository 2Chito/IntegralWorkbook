\begin{newmondai}{例題}{4}{今週の積分}{*}
    \int_1^2 \frac{1}{2^x} dx
\end{newmondai}

\approachitem{$\int a^x dx = \frac{a^x}{\log a}+\mathbb{C}$}

\solutionhead
\hfill
以下に計算手順を示す.
\hfill\
\begin{align*}
    \left(2^{-x}\right)'=-2^{-x}\log 2
\end{align*}
より、
\begin{align*}
    \left(-\frac{2^{-x}}{\log 2}\right)'=2^{-x}
\end{align*}
\begin{align*}
    \therefore \int_1^2 \frac{1}{2^x}dx
        &= \int_1^2 2^{-x}dx \\
        &= \left[ -\frac{2^{-x}}{\log 2}\right]_1^2 \\
        &= -\frac{1}{\log 2} \left(2^{-2}-2^{-1}\right) \\
        &= \boldsymbol{\frac{1}{4\log 2}}
\end{align*}
