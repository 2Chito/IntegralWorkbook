    \begin{pointbox}

        \end{center}

    % --- ここから解説パート（下半分） ---
    \begin{align*}
        \int \tan ^3 x dx   &= \int \tan x \cdot \tan x ^2 x dx \\
                            &= \int \tan x \left( \frac{1}{\cos ^2 x} -1 \right)  dx \\
                            &= \int \tan x \cdot \frac{1}{\cos ^2 x} dx + \int \frac{-\sin x}{\cos x} dx \\
                            &= \frac{1}{2} \tan ^2 x + \log \left| \cos x \right| + \mathbb{C} \tag{答}
    \end{align*}
\end{mondai}
