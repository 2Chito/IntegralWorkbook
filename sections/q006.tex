\approachitem{$\sqrt{1+x^2}\text{を含む形}\rightarrow x=\frac{e^t-e^{-t}}{2} \text{と置換}$}

\solutionhead
\hfill
以下に計算手順を示す.
\hfill\
\Itembox{
    x=\frac{e^t-e^{-t}}{2} \qquad
    t=\log \left(x+\sqrt{x^2+1}\right) \qquad 
    dx=\frac{e^t+e^{-t}}{2}dt
}
\begin{align*}
    \int \frac{1}{\sqrt{1+x^2}} dx
        &= \int \frac{1}{\sqrt{1+\frac{\left(e^t-e^{-t}\right)^2}{4}}} \frac{e^t+e^{-t}}{2}dt \\
        &= \int \frac{1}{\frac{e^t+e^{-t}}{2}}\frac{e^t+e^{-t}}{2} dt \\
        &= \int dt \\
        &= t+\mathbb{C} \\
        &= \boldsymbol{\log \left(x+\sqrt{x^2+1}\right)+\mathbb{C}}
\end{align*}
